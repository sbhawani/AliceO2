38 #define makelogaxis(h) \
39  { \
40  const Int_t nbins = h->GetNbinsX(); \
41  double binp[nbins + 1]; \
42  double max = h->GetXaxis()->GetBinUpEdge(nbins); \
43  double min = h->GetXaxis()->GetBinLowEdge(1); \
44  double lmin = TMath::Log10(min); \
45  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins); \
46  for (int i = 0; i < nbins; i++) { \
47  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta)); \
48  } \
49  binp[nbins] = max + 1; \
50  h->GetXaxis()->Set(nbins, binp); \
51  }

53 constexpr int Np = 9;
54 struct TPCSpectraTask {
55  static constexpr const char* pT[Np] = {"e", "#mu", "#pi", "K", "p", "d", "t", "^{3}He", "#alpha"};
56  static constexpr const char* hp[Np] = {"p/El", "p/Mu", "p/Pi", "p/Ka", "p/Pr", "p/De", "p/Tr", "p/He", "p/Al"};
57  static constexpr const char* hpt[Np] = {"pt/El", "pt/Mu", "pt/Pi", "pt/Ka", "pt/Pr", "pt/De", "pt/Tr", "pt/He", "pt/Al"};
58  HistogramRegistry histos{"Histos", {}, OutputObjHandlingPolicy::AnalysisObject};
59 
60  void init(o2::framework::InitContext&)
61  {
62  histos.add("p/Unselected", "Unselected;#it{p} (GeV/#it{c})", kTH1F, {{100, 0, 20}});
63  histos.add("pt/Unselected", "Unselected;#it{p}_{T} (GeV/#it{c})", kTH1F, {{100, 0, 20}});
64  for (int i = 0; i < Np; i++) {
65  histos.add(hp[i], Form("%s;#it{p} (GeV/#it{c})", pT[i]), kTH1F, {{100, 0, 20}});
66  histos.add(hpt[i], Form("%s;#it{p}_{T} (GeV/#it{c})", pT[i]), kTH1F, {{100, 0, 20}});
67  }
68  }
69 
70  //Defining filters and input
71  CANDIDATE_SELECTION
72 
73  Configurable<float> nsigmacut{"nsigmacut", 3, "Value of the Nsigma cut"};
74 
75  using TrackCandidates = soa::Filtered<soa::Join<aod::Tracks, aod::TracksExtra, aod::pidRespTPC, aod::TrackSelection>>;
76  void process(TrackCandidates::iterator const& track)
77  {
78  const float nsigma[Np] = {track.tpcNSigmaEl(), track.tpcNSigmaMu(), track.tpcNSigmaPi(),
79  track.tpcNSigmaKa(), track.tpcNSigmaPr(), track.tpcNSigmaDe(),
80  track.tpcNSigmaTr(), track.tpcNSigmaHe(), track.tpcNSigmaAl()};
81  histos.fill("p/Unselected", track.p());
82  histos.fill("pt/Unselected", track.pt());
83  for (int i = 0; i < Np; i++) {
84  if (abs(nsigma[i]) > nsigmacut.value) {
85  continue;
86  }
87  histos.fill(hp[i], track.p());
88  histos.fill(hpt[i], track.pt());
89  }
90  }
91 };
