213 template <typename T>
214 void IntervalXY<T>::getLineCoefs(T& a, T& b, T& c) const
215 {
216  // convert to line parameters in canonical form: a*x+b*y+c = 0
217  c = mX * mDY - mY * mDx;
218  if (c) {
219  a = -mDY;
220  b = mDx;
221  } else if (fabs(mDx) > fabs(mDY)) {
222  a = 0.;
223  b = -1.;
224  c = mY;
225  } else {
226  a = -1.;
227  b = 0.;
228  c = mX;
229  }
230 }

232 template <typename T>
233 bool IntervalXY<T>::seenByCircle(const CircleXY<T>& circle, T eps) const
234 {
235  T dx0 = mX - circle.xC;
236  T dy0 = mY - circle.yC;
237  T dx1 = dx0 + mDx;
238  T dy1 = dy0 + mDY;
239  if (eps > 0.) {
240  const T dxeps = eps * mDx, dyeps = eps * mDY;
241  dx0 -= dxeps;
242  dx1 += dxeps;
243  dy0 -= dyeps;
244  dy1 += dyeps;
245  }
246  // distance^2 from circle center to edges
247  const T d02 = dx0 * dx0 + dy0 * dy0;
248  const T d12 = dx1 * dx1 + dy1 * dy1;
249  const T rC2 = circle.rC * circle.rC;
250 
251  return (d02 - rC2) * (d12 - rC2) < 0;
252 }
253 template <typename T>
254 bool IntervalXY<T>::circleCrossParam(const CircleXY<T>& circle, T& t) const
255 {
256  const T dx = mX - circle.xC;
257  const T dy = mY - circle.yC;
258  const T del2I = 1.f / (mDx * mDx + mDY * mDY);
259  const T b = (dx * mDx + dy * mDY) * del2I;
260  const T c = del2I * (dx * dx + dy * dy - circle.rC * circle.rC);
261  T det = b * b - c;
262  if (det < 0.f) {
263  return false;
264  }
265  det = std::sqrt(det);
266  const T t0 = -b - det;
267  const T t1 = -b + det;
268  // select the one closer to [0:1] interval
269  t = (std::fabs(t0 - 0.5f) < std::fabs(t1 - 0.5f)) ? t0 : t1;
270  return true;
271 }

273 template <typename T>
274 bool IntervalXY<T>::seenByLine(const IntervalXY<T>& other, T eps) const
275 {
276  T a, b, c; // find equation of the line a*x+b*y+c = 0
277  other.getLineCoefs(a, b, c);
278  const auto [x0, y0] = eval(-eps);
279  const auto [x1, y1] = eval(1.f + eps);
280 
281  return (a * x0 + b * y0 + c) * (a * x1 + b * y1 + c) < 0;
282 }

284 template <typename T>
285 bool IntervalXY<T>::lineCrossParam(const IntervalXY<T>& other, T& t) const
286 {
287  // tolerance
288  constexpr float eps = 1.e-9f;
289  const T dx = other.getX0() - mX;
290  const T dy = other.getY0() - mY;
291  const T det = -mDx * other.getY0() + mDY * other.getX0();
292  if (std::fabs(det) < eps) {
293  return false; // parallel
294  }
295  t = (-dx * other.getY0() + dy * other.getX0()) / det;
296  return true;
297 }
