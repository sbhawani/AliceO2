41 template <typename value_T>
42 void g3helx3(value_T qfield, value_T step, std::array<value_T, 7>& vect)
43 {
44  /******************************************************************
45  * *
46  * GEANT3 tracking routine in a constant field oriented *
47  * along axis 3 *
48  * Tracking is performed with a conventional *
49  * helix step method *
50  * *
51  * Authors R.Brun, M.Hansroul ********* *
52  * Rewritten V.Perevoztchikov *
53  * *
54  * Rewritten in C++ by I.Belikov *
55  * *
56  * qfield (kG) - particle charge times magnetic field *
57  * step (cm) - step length along the helix *
58  * vect[7](cm,GeV/c) - input/output x, y, z, px/p, py/p ,pz/p, p *
59  * *
60  ******************************************************************/
61 
62  static_assert(std::is_floating_point_v<value_T>);
63 
64  const int ix = 0, iy = 1, iz = 2, ipx = 3, ipy = 4, ipz = 5, ipp = 6;
65  constexpr value_T kOvSqSix = 0.408248f; // std::sqrt(1./6.);
66 
67  value_T cosx = vect[ipx], cosy = vect[ipy], cosz = vect[ipz];
68 
69  value_T rho = qfield * constants::math::B2C / vect[ipp];
70  value_T tet = rho * step;
71 
72  value_T tsint, sintt, sint, cos1t;
73  if (std::fabs(tet) > 0.03f) {
74  sint = std::sin(tet);
75  sintt = sint / tet;
76  tsint = (tet - sint) / tet;
77  value_T t = std::sin(0.5f * tet);
78  cos1t = 2 * t * t / tet;
79  } else {
80  tsint = tet * tet / 6.f;
81  sintt = (1.f - tet * kOvSqSix) * (1.f + tet * kOvSqSix); // 1.- tsint;
82  sint = tet * sintt;
83  cos1t = 0.5f * tet;
84  }
85 
86  value_T f1 = step * sintt;
87  value_T f2 = step * cos1t;
88  value_T f3 = step * tsint * cosz;
89  value_T f4 = -tet * cos1t;
90  value_T f5 = sint;
91 
92  vect[ix] += f1 * cosx - f2 * cosy;
93  vect[iy] += f1 * cosy + f2 * cosx;
94  vect[iz] += f1 * cosz + f3;
95 
96  vect[ipx] += f4 * cosx - f5 * cosy;
97  vect[ipy] += f4 * cosy + f5 * cosx;
98 }

101 template <typename value_T>
102 value_T BetheBlochSolid(value_T bg, value_T rho, value_T kp1, value_T kp2, value_T meanI,
103  value_T meanZA)
104 {
105  //
106  // This is the parameterization of the Bethe-Bloch formula inspired by Geant.
107  //
108  // bg - beta*gamma
109  // rho - density [g/cm^3]
110  // kp1 - density effect first junction point
111  // kp2 - density effect second junction point
112  // meanI - mean excitation energy [GeV]
113  // meanZA - mean Z/A
114  //
115  // The default values for the kp* parameters are for silicon.
116  // The returned value is in [GeV/(g/cm^2)].
117  //
118  static_assert(std::is_floating_point_v<value_T>);
119 
120  constexpr value_T mK = 0.307075e-3f; // [GeV*cm^2/g]
121  constexpr value_T me = 0.511e-3f; // [GeV/c^2]
122  kp1 *= 2.303f;
123  kp2 *= 2.303f;
124  value_T bg2 = bg * bg;
125  value_T maxT = 2.f * me * bg2; // neglecting the electron mass
126 
127  //*** Density effect
128  value_T d2 = 0.;
129  const value_T x = std::log(bg);
130  const value_T lhwI = std::log(28.816f * 1e-9f * std::sqrt(rho * meanZA) / meanI);
131  if (x > kp2) {
132  d2 = lhwI + x - 0.5f;
133  } else if (x > kp1) {
134  double r = (kp2 - x) / (kp2 - kp1);
135  d2 = lhwI + x - 0.5f + (0.5f - lhwI - kp1) * r * r * r;
136  }
137  return mK * meanZA * (1 + bg2) / bg2 * (0.5f * std::log(2 * me * bg2 * maxT / (meanI * meanI)) - bg2 / (1 + bg2) - d2);
138 }
