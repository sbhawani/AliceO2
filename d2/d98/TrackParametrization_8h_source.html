16 /*
17  24/09/2020: Added new data member for abs. charge. This is needed for uniform treatment of tracks with non-standard
18  charge: 0 (for V0s) and e.g. 2 for hypernuclei.
19  In the aliroot AliExternalTrackParam this was treated by derived classes using virtual methods, which we don't use in O2.
20  The meaning of mP[kQ2Pt] remains exactly the same, except for q=0 case: in this case the mP[kQ2Pt] is just an alias to
21  1/pT, regardless of its sign, and the getCurvature() will 0 (because mAbsCharge is 0).
22  The methods returning lab momentum or its combination account for eventual q>1.

417 template <typename value_T>
418 inline void TrackParametrization<value_T>::getCircleParamsLoc(value_t bz, o2::math_utils::CircleXY<value_t>& c) const
419 {
420  // get circle params in track local frame, for straight line just set to local coordinates
421  c.rC = getCurvature(bz);
422  // treat as straight track if sagitta between the vertex and middle of TPC is below 0.01 cm
423  constexpr value_t MinSagitta = 0.01f, TPCMidR = 160.f, MinCurv = 8 * MinSagitta / (TPCMidR * TPCMidR);
424  if (std::abs(c.rC) > MinCurv) {
425  c.rC = 1.f / getCurvature(bz);
426  value_t sn = getSnp(), cs = std::sqrt((1.f - sn) * (1.f + sn));
427  c.xC = getX() - sn * c.rC; // center in tracking
428  c.yC = getY() + cs * c.rC; // frame. Note: r is signed!!!
429  c.rC = std::abs(c.rC);
430  } else {
431  c.rC = 0.f; // signal straight line
432  c.xC = getX();
433  c.yC = getY();
434  }
435 }

438 template <typename value_T>
439 inline void TrackParametrization<value_T>::getCircleParams(value_t bz, o2::math_utils::CircleXY<value_t>& c, value_t& sna, value_t& csa) const
440 {
441  // get circle params in loc and lab frame, for straight line just set to global coordinates
442  getCircleParamsLoc(bz, c);
443  o2::math_utils::detail::sincos(getAlpha(), sna, csa);
444  o2::math_utils::detail::rotateZ<value_t>(c.xC, c.yC, c.xC, c.yC, sna, csa); // center in global frame
445 }

448 template <typename value_T>
449 inline void TrackParametrization<value_T>::getLineParams(o2::math_utils::IntervalXY<value_t>& ln, value_t& sna, value_t& csa) const
450 {
451  // get line parameterization as { x = x0 + xSlp*t, y = y0 + ySlp*t }
452  o2::math_utils::detail::sincos(getAlpha(), sna, csa);
453  o2::math_utils::detail::rotateZ<value_t>(getX(), getY(), ln.getX0(), ln.getY0(), sna, csa); // reference point in global frame
454  value_t snp = getSnp(), csp = std::sqrt((1.f - snp) * (1.f + snp));
455  ln.setDX(csp * csa - snp * sna);
456  ln.setDY(snp * csa + csp * sna);
457 }

509 //____________________________________________________________
510 template <typename value_T>
511 inline typename TrackParametrization<value_T>::value_t TrackParametrization<value_T>::getP2Inv() const
512 {
513  // return the inverted track momentum^2
514  const value_t p2 = mP[kQ2Pt] * mP[kQ2Pt] / (1.f + getTgl() * getTgl());
515  return (mAbsCharge > 1) ? p2 * mAbsCharge * mAbsCharge : p2;
516 }

521 {
522  // return the track momentum^2
523  const value_t p2inv = getP2Inv();
524  return (p2inv > o2::constants::math::Almost0) ? 1.f / p2inv : o2::constants::math::VeryBig;
525 }

528 template <typename value_T>

530 {

532  const value_t pInv = std::fabs(mP[kQ2Pt]) / std::sqrt(1.f + getTgl() * getTgl());
533  return (mAbsCharge > 1) ? pInv / mAbsCharge : pInv;
534 }

573 inline math_utils::Point3D<typename TrackParametrization<value_T>::value_t> TrackParametrization<value_T>::getXYZGlo() const

582 }

595 //_______________________________________________________
596 template <typename value_T>
597 inline math_utils::Point3D<typename TrackParametrization<value_T>::value_t> TrackParametrization<value_T>::getXYZGloAt(value_t xk, value_t b, bool& ok) const
598 {
599  //----------------------------------------------------------------
600  // estimate global X,Y,Z in global frame at given X
601  //----------------------------------------------------------------
602  value_t y = 0.f, z = 0.f;
603  ok = getYZAt(xk, b, y, z);
604  if (ok) {
605 #ifndef GPUCA_ALIGPUCODE
606  return math_utils::Rotation2D<value_t>(getAlpha())(math_utils::Point3D<value_t>(xk, y, z));
607 #else // mockup on GPU without ROOT
608  float sina, cosa;
609  o2::gpu::CAMath::SinCos(getAlpha(), sina, cosa);
610  return math_utils::Point3D<value_t>(cosa * xk + sina * y, cosa * y - sina * xk, z);
611 #endif
612  } else {
613  return math_utils::Point3D<value_t>();
614  }
615 }
