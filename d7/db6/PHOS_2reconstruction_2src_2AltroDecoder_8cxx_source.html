61 void AltroDecoder::readChannels()
62 {
63  mChannelsInitialized = false;
64  mChannels.clear();
65  int currentpos = 0;
66  auto& buffer = mRawReader.getPayload().getPayloadWords();
67  while (currentpos < buffer.size() - mRCUTrailer.getTrailerSize()) {
68  auto currentword = buffer[currentpos++];
69  if (currentword >> 30 != 1) {
70  continue;
71  }
72  // starting a new channel
73  mChannels.emplace_back(currentword & 0xFFF, (currentword >> 16) & 0x3FF);
74  auto& currentchannel = mChannels.back();
76  int numberofwords = (currentchannel.getPayloadSize() + 2) / 3;
77  std::vector<uint16_t> bunchwords;
78  for (int iword = 0; iword < numberofwords; iword++) {
79  currentword = buffer[currentpos++];
80  if ((currentword >> 30) != 0) {
81  LOG(ERROR) << "Unexpected end of payload in altro channel payload! FEE=" << o2::raw::RDHUtils::getFEEID(mRawReader.getRawHeader())
82  << ", Address=0x" << std::hex << currentchannel.getHardwareAddress() << ", word=0x" << currentword << std::dec;
83  currentpos--;
84  continue;
85  }
86  bunchwords.push_back((currentword >> 20) & 0x3FF);
87  bunchwords.push_back((currentword >> 10) & 0x3FF);
88  bunchwords.push_back(currentword & 0x3FF);
89  }
90 
91  // decode bunches
92  int currentsample = 0;
93  while (currentsample < currentchannel.getPayloadSize()) {
94  int bunchlength = bunchwords[currentsample] - 2, // remove words for bunchlength and starttime
95  starttime = bunchwords[currentsample + 1];
96  auto& currentbunch = currentchannel.createBunch(bunchlength, starttime);
97  currentbunch.initFromRange(gsl::span<uint16_t>(&bunchwords[currentsample + 2], std::min((unsigned long)bunchlength, bunchwords.size() - currentsample - 2)));
98  currentsample += bunchlength + 2;
99  }
100  }
101  mChannelsInitialized = true;
102 }
