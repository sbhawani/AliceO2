127  void process(MyEvents::iterator const& collision, aod::MuonClusters const& clustersMuon, aod::Muons const& tracksMuon, aod::BCs const& bcs, MyBarrelTracks const& tracksBarrel)
128  {
129  uint64_t tag = 0;
130  uint32_t triggerAliases = 0;
131  for (int i = 0; i < kNaliases; i++) {
132  if (collision.alias()[i] > 0) {
133  triggerAliases |= (uint32_t(1) << i);
134  }
135  }
136 
137  VarManager::ResetValues(0, VarManager::kNEventWiseVariables, fValues);
138  VarManager::FillEvent<gkEventFillMap>(collision, fValues); // extract event information and place it in the fgValues array
139  fHistMan->FillHistClass("Event_BeforeCuts", fValues); // automatically fill all the histograms in the class Event
140 
141  if (!fEventCut->IsSelected(fValues)) {
142  return;
143  }
144 
145  fHistMan->FillHistClass("Event_AfterCuts", fValues);
146 
147  event(tag, collision.bc().runNumber(), collision.posX(), collision.posY(), collision.posZ(), collision.numContrib());
148  eventExtended(collision.bc().globalBC(), collision.bc().triggerMask(), triggerAliases, 0.0f);
149  eventVtxCov(collision.covXX(), collision.covXY(), collision.covXZ(), collision.covYY(), collision.covYZ(), collision.covZZ(), collision.chi2());
150 
151  uint64_t trackFilteringTag = 0;
152  trackBasic.reserve(barrelSelectedTracks.size());
153  trackBarrel.reserve(barrelSelectedTracks.size());
154  trackBarrelCov.reserve(barrelSelectedTracks.size());
155  trackBarrelPID.reserve(barrelSelectedTracks.size());
156 
157  for (auto& track : barrelSelectedTracks) {
158  VarManager::FillTrack<gkTrackFillMap>(track, fValues);
159  fHistMan->FillHistClass("TrackBarrel_BeforeCuts", fValues);
160  if (!fTrackCut->IsSelected(fValues)) {
161  continue;
162  }
163  fHistMan->FillHistClass("TrackBarrel_AfterCuts", fValues);
164 
165  if (track.isGlobalTrack()) {
166  trackFilteringTag |= (uint64_t(1) << 0);
167  }
168  if (track.isGlobalTrackSDD()) {
169  trackFilteringTag |= (uint64_t(1) << 1);
170  }
171  trackBasic(event.lastIndex(), track.globalIndex(), trackFilteringTag, track.pt(), track.eta(), track.phi(), track.charge());
172  trackBarrel(track.tpcInnerParam(), track.flags(), track.itsClusterMap(), track.itsChi2NCl(),
173  track.tpcNClsFindable(), track.tpcNClsFindableMinusFound(), track.tpcNClsFindableMinusCrossedRows(),
174  track.tpcNClsShared(), track.tpcChi2NCl(),
175  track.trdChi2(), track.tofChi2(),
176  track.length(), track.dcaXY(), track.dcaZ());
177  trackBarrelCov(track.cYY(), track.cZZ(), track.cSnpSnp(), track.cTglTgl(), track.c1Pt21Pt2());
178  trackBarrelPID(track.tpcSignal(),
179  track.tpcNSigmaEl(), track.tpcNSigmaMu(),
180  track.tpcNSigmaPi(), track.tpcNSigmaKa(), track.tpcNSigmaPr(),
181  track.tpcNSigmaDe(), track.tpcNSigmaTr(), track.tpcNSigmaHe(), track.tpcNSigmaAl(),
182  track.tofSignal(), track.beta(),
183  track.tofNSigmaEl(), track.tofNSigmaMu(),
184  track.tofNSigmaPi(), track.tofNSigmaKa(), track.tofNSigmaPr(),
185  track.tofNSigmaDe(), track.tofNSigmaTr(), track.tofNSigmaHe(), track.tofNSigmaAl(),
186  track.trdSignal());
187  }
188 
189  muonBasic.reserve(muonSelectedTracks.size());
190  muonExtended.reserve(muonSelectedTracks.size());
191  for (auto& muon : muonSelectedTracks) {
192  // TODO: add proper information for muon tracks
193  if (muon.bcId() != collision.bcId()) {
194  continue;
195  }
196  // TODO: the trackFilteringTag will not be needed to encode whether the track is a muon since there is a dedicated table for muons
197  trackFilteringTag |= (uint64_t(1) << 0); // this is a MUON arm track
198  muonBasic(event.lastIndex(), trackFilteringTag, muon.pt(), muon.eta(), muon.phi(), muon.charge());
199  muonExtended(muon.inverseBendingMomentum(), muon.thetaX(), muon.thetaY(), muon.zMu(), muon.bendingCoor(), muon.nonBendingCoor(), muon.chi2(), muon.chi2MatchTrigger());
200  }
201  }

203  void DefineHistograms(TString histClasses)
204  {
205  const int kNRuns = 2;
206  int runs[kNRuns] = {244918, 244919};
207  TString runsStr;
208  for (int i = 0; i < kNRuns; i++) {
209  runsStr += Form("%d;", runs[i]);
210  }
211  VarManager::SetRunNumbers(kNRuns, runs);
212 
213  std::unique_ptr<TObjArray> arr(histClasses.Tokenize(";"));
214  for (Int_t iclass = 0; iclass < arr->GetEntries(); ++iclass) {
215  TString classStr = arr->At(iclass)->GetName();
216 
217  if (classStr.Contains("Event")) {
218  fHistMan->AddHistClass(classStr.Data());
219  fHistMan->AddHistogram(classStr.Data(), "VtxZ", "Vtx Z", false, 60, -15.0, 15.0, VarManager::kVtxZ); // TH1F histogram
220  fHistMan->AddHistogram(classStr.Data(), "VtxZ_Run", "Vtx Z", true,
221  kNRuns, 0.5, 0.5 + kNRuns, VarManager::kRunId, 60, -15.0, 15.0, VarManager::kVtxZ, 10, 0., 0., VarManager::kNothing, runsStr.Data()); // TH1F histogram
222  fHistMan->AddHistogram(classStr.Data(), "VtxX_VtxY", "Vtx X vs Vtx Y", false, 100, 0.055, 0.08, VarManager::kVtxX, 100, 0.31, 0.35, VarManager::kVtxY); // TH2F histogram
223  fHistMan->AddHistogram(classStr.Data(), "VtxX_VtxY_VtxZ", "vtx x - y - z", false, 100, 0.055, 0.08, VarManager::kVtxX, 100, 0.31, 0.35, VarManager::kVtxY, 60, -15.0, 15.0, VarManager::kVtxZ); // TH3F histogram
224  fHistMan->AddHistogram(classStr.Data(), "NContrib_vs_VtxZ_prof", "Vtx Z vs ncontrib", true, 30, -15.0, 15.0, VarManager::kVtxZ, 10, -1., 1., VarManager::kVtxNcontrib); // TProfile histogram
225  fHistMan->AddHistogram(classStr.Data(), "VtxZ_vs_VtxX_VtxY_prof", "Vtx Z vs (x,y)", true, 100, 0.055, 0.08, VarManager::kVtxX, 100, 0.31, 0.35, VarManager::kVtxY, 10, -1., 1., VarManager::kVtxZ); // TProfile2D histogram
226  fHistMan->AddHistogram(classStr.Data(), "Ncontrib_vs_VtxZ_VtxX_VtxY_prof", "n-contrib vs (x,y,z)", true,
227  100, 0.055, 0.08, VarManager::kVtxX, 100, 0.31, 0.35, VarManager::kVtxY, 30, -15., 15., VarManager::kVtxZ,
228  "", "", "", VarManager::kVtxNcontrib); // TProfile3D
229  }
230 
231  if (classStr.Contains("Track")) {
232  fHistMan->AddHistClass(classStr.Data());
233  fHistMan->AddHistogram(classStr.Data(), "Pt", "p_{T} distribution", false, 200, 0.0, 20.0, VarManager::kPt); // TH1F histogram
234  fHistMan->AddHistogram(classStr.Data(), "Eta", "#eta distribution", false, 500, -5.0, 5.0, VarManager::kEta); // TH1F histogram
235  fHistMan->AddHistogram(classStr.Data(), "Phi_Eta", "#phi vs #eta distribution", false, 200, -5.0, 5.0, VarManager::kEta, 200, -6.3, 6.3, VarManager::kPhi); // TH2F histogram
236 
237  if (classStr.Contains("Barrel")) {
238  fHistMan->AddHistogram(classStr.Data(), "TPCncls", "Number of cluster in TPC", false, 160, -0.5, 159.5, VarManager::kTPCncls); // TH1F histogram
239  fHistMan->AddHistogram(classStr.Data(), "TPCncls_Run", "Number of cluster in TPC", true, kNRuns, 0.5, 0.5 + kNRuns, VarManager::kRunId,
240  10, -0.5, 159.5, VarManager::kTPCncls, 10, 0., 1., VarManager::kNothing, runsStr.Data()); // TH1F histogram
241  fHistMan->AddHistogram(classStr.Data(), "ITSncls", "Number of cluster in ITS", false, 8, -0.5, 7.5, VarManager::kITSncls); // TH1F histogram
242  //for TPC PID
243  fHistMan->AddHistogram(classStr.Data(), "TPCdedx_pIN", "TPC dE/dx vs pIN", false, 200, 0.0, 20.0, VarManager::kPin, 200, 0.0, 200., VarManager::kTPCsignal); // TH2F histogram
244  fHistMan->AddHistogram(classStr.Data(), "DCAxy", "DCAxy", false, 100, -3.0, 3.0, VarManager::kTrackDCAxy); // TH1F histogram
245  fHistMan->AddHistogram(classStr.Data(), "DCAz", "DCAz", false, 100, -5.0, 5.0, VarManager::kTrackDCAz); // TH1F histogram
246  fHistMan->AddHistogram(classStr.Data(), "IsGlobalTrack", "IsGlobalTrack", false, 2, -0.5, 1.5, VarManager::kIsGlobalTrack); // TH1F histogram
247  fHistMan->AddHistogram(classStr.Data(), "IsGlobalTrackSDD", "IsGlobalTrackSDD", false, 2, -0.5, 1.5, VarManager::kIsGlobalTrackSDD); // TH1F histogram
248  }
249  }
250  } // end loop over histogram classes
251  }
