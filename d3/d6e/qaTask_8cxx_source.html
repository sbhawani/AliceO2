37 double ConvertPhiRange(double phi)
38 {
39  if (phi > M_PI) {
40  phi -= 2 * M_PI;
41  } else if (phi < -M_PI) {
42  phi += 2 * M_PI;
43  }
44 
45  return phi;
46 }

55 template <typename Track>
56 bool GetImpactParameterAndError(const Track& track, const o2df::VertexBase& primaryVertex, double& impactParameterRPhi,
57  double& impactParameterRPhiError, double& impactParameterZ,
58  double& impactParameterErrorZ)
59 {
60  impactParameterRPhi = -999.;
61  impactParameterRPhiError = -999.;
62  impactParameterZ = -999;
63  impactParameterErrorZ = -999;
64 
65  o2df::DCA dca;
66  // FIXME: get this from CCDB
67  constexpr float magneticField{5.0}; // in kG
68  auto trackParameter = getTrackParCov(track);
69  bool propagate = trackParameter.propagateToDCA(primaryVertex, magneticField, &dca);
70 
71  constexpr float conversion_to_micrometer = 1000;
72  if (propagate) {
73  impactParameterRPhi = conversion_to_micrometer * dca.getY();
74  impactParameterRPhiError = conversion_to_micrometer * std::sqrt(dca.getSigmaY2());
75  impactParameterZ = conversion_to_micrometer * dca.getZ();
76  impactParameterErrorZ = conversion_to_micrometer * std::sqrt(dca.getSigmaZ2());
77  }
78  return propagate;
79 }

201  void process(const o2::aod::Collision& collision, const o2::aod::Tracks& tracks)
202  {

206  histograms.fill("collision/collisionZ", collision.posZ());
207 
208  int nTracks(0);
209  for (const auto& track : tracks) {
210  nTracks++;
211  }
212 
213  histograms.fill("multiplicity/numberOfTracks", nTracks);
214 
215  // fill covariance variables
216  histograms.fill("Covariance/xx", collision.covXX());
217  histograms.fill("Covariance/xy", collision.covXY());
218  histograms.fill("Covariance/xz", collision.covXZ());
219  histograms.fill("Covariance/yy", collision.covYY());
220  histograms.fill("Covariance/yz", collision.covYZ());
221  histograms.fill("Covariance/zz", collision.covZZ());
222 
223  // fill quality variables
224  histograms.fill("Quality/Chi2", collision.chi2());
225  histograms.fill("Quality/Contributors", collision.numContrib());

388  void process(const o2::soa::Join<o2::aod::Collisions, o2::aod::McCollisionLabels>::iterator& collision,
389  const o2::soa::Join<o2::aod::Tracks, o2::aod::TracksCov, o2::aod::McTrackLabels>& tracks,
390  const o2::aod::McParticles& mcParticles, const o2::aod::McCollisions& mcCollisions)
391  {
392  const o2df::VertexBase primaryVertex = getPrimaryVertex(collision);
393 
394  for (const auto& track : tracks) {
395  const double deltaPt = track.label().pt() - track.pt();
396  histos.fill("pt/ptDiffMCRec", deltaPt);
397 
398  const double deltaPtOverPt = deltaPt / track.pt();
399 
400  histos.fill("pt/ptResolution", deltaPtOverPt);
401  histos.fill("pt/ptResolutionVsPt", track.pt(), std::abs(deltaPtOverPt));
402  histos.fill("pt/ptResolutionVsEta", track.eta(), std::abs(deltaPtOverPt));
403  histos.fill("pt/ptResolutionVsPhi", track.phi(), std::abs(deltaPtOverPt));
404 
405  const double deltaEta = track.label().eta() - track.eta();
406  histos.fill("eta/etaDiffMCReco", deltaEta);
407  histos.fill("eta/etaDiffMCRecoVsEtaMC", deltaEta, track.label().eta());
408  histos.fill("eta/etaDiffMCRecoVsEtaReco", deltaEta, track.eta());
409 
410  const double deltaPhi = track_utils::ConvertPhiRange(track.label().phi() - track.phi());
411  histos.fill("phi/phiDiffMCRec", deltaPhi);
412 
413  double impactParameterRPhi{-999.}, impactParameterRPhiError{-999.};
414  double impactParameterZ{-999.}, impactParameterErrorZ{-999.};
415 
416  const bool propagate = track_utils::GetImpactParameterAndError(
417  track, primaryVertex, impactParameterRPhi, impactParameterRPhiError, impactParameterZ, impactParameterErrorZ);
418 
419  if (propagate) {
420  histos.fill("impactParameter/impactParameterRPhiVsPt", track.pt(), impactParameterRPhi);
421  histos.fill("impactParameter/impactParameterRPhiVsEta", track.eta(), impactParameterRPhi);
422  histos.fill("impactParameter/impactParameterRPhiVsPhi", track.phi(), impactParameterRPhi);
423 
424  histos.fill("impactParameter/impactParameterZVsPt", track.pt(), impactParameterZ);
425  histos.fill("impactParameter/impactParameterZVsEta", track.eta(), impactParameterZ);
426  histos.fill("impactParameter/impactParameterZVsPhi", track.phi(), impactParameterZ);
427 
428  histos.fill("impactParameter/impactParameterErrorRPhiVsPt", track.pt(), impactParameterRPhiError);
429  histos.fill("impactParameter/impactParameterErrorRPhiVsEta", track.eta(), impactParameterRPhiError);
430  histos.fill("impactParameter/impactParameterErrorRPhiVsPhi", track.phi(), impactParameterRPhiError);
431 
432  histos.fill("impactParameter/impactParameterErrorZVsPt", track.pt(), impactParameterErrorZ);
433  histos.fill("impactParameter/impactParameterErrorZVsEta", track.eta(), impactParameterErrorZ);
434  histos.fill("impactParameter/impactParameterErrorZVsPhi", track.phi(), impactParameterErrorZ);
